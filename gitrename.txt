Базовые гид команды
git init - инициал. локального репоз.
git add { либо имя ффайла либо поставить просто точку добавление файла в индексанцию}  
git rm filename --cached - удаляет файл из индексации
git commit -m "сообщение"
git log
git branch просмотр всех веток
git barnch <name> - создание новой ветки
git checkout <имя ветки>
git merge имя ветки - вливание в текущую ветку указанной ветки
git branch -d <имя ветки> - удаление ветки
